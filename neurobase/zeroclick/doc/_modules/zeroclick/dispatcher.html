#  dispatcher.py
#
#     input: none
#
#     action:  loops forever, sleeping for SLEEP_INTERVAL:
#               -- finds new projects in assembled data directory, PIPELINE_PATH,
#                  and adds annotation jobs to queue table in zero_click database.
#               -- looks for completed projects in queue table and adds them to
#                  completed_projects table, and if they have a job_type='upload',
#                  ftp's the relevant files to Whitney's ftp server.
#
#     ====================================================================================
#     BEFORE USING: modify SLEEP_INTERVAL, PIPELINE_PATH, (possibly) REMOTE_PIPELINE_PATH,
#                   and change "_dev" ==> "" globally in this file.
#     ====================================================================================
#
#     details:
#       open database tables: queue, pn_mapping, jn_mapping, configuration,
#                             default_configuration, init_projects, completed_projects,
#                             submitted_projects, args, default_args
#       Default Jobs for a Project: uses entries in default_configuration & default_args tables
#       Special Case: uses entries in pn_mapping, jn_mapping, configuration, & args tables.
#
#       loop forever:
#         ENTER NEW PROJECTS INTO QUEUE:
#         read PIPELINE_PATH dir to find all projects to be annotated,
#         foreach <project_name> found:
#           look up <project_name> in pn_mapping_table
#           if there ==> not a default job; so ==> <project_id>
#             if <project_id> not in init_projects_table:
#               insert <project_id> into init_projects_table
#               foreach <project_id> entry in configuration_table ==> <job_type>, <code>
#                 if <code> == '+':
#                   create <job_name> = <project_name>_<job type>
#	            lookup <job_type> & <project_id> in jn_mapping table ==> <job_id>
#                   lookup <job_id> in args table ==> <args_list>
#                   insert <project_id>, <job_id>, <job_type>, <args_list> into queue table
#                      setting started, finished = 'N'
#               insert <project_id> into submitted_projects table
#           else: <project_name> not in pn_table so its a default job
#             insert <project_name> into pn_mapping -> <project_id>
#             insert <project_id> into init_projects table
#             foreach <job_type> in default_configuration:
#               if <code> == '+':
#                 create <job_name> = <project_name>_<job type>
#                 insert <job_type>,<project_id> into jn_mapping -> <job_id>
#                 lookup <job_type> in default_args table -> <args_list>
#                 insert <project_id>, <job_id>, <job_type>, <args_list> into queue table
#                      setting started, finished = 'N'
#             insert <project_id> into submitted_projects table
#
#         PROCESS COMPLETED JOBS
#         find all submitted projects, not in completed projects ==> <project_id_list>
#         foreach <project_id> in <project_id_list>:
#           find all jobs for that <project_id> in queue_table where finished = 'N'
#           if no rows with finished=N, or one row with finished=N and job_type = upload then proj is complete
#           if found job_type=upload and all other rows finished:
#             do upload:
#               get project_name from pn_mapping_table
#               ftp = FTP('128.227.123.35','ftpguest','7002ram')
#               send all files in PIPELINE_PATH/<project_name> dir
#               to ftp.server/REMOTE_PIPELINE_PATH/<project_name>
#         insert project_id in completed_projects
#         if job_type='upload' set upload finished='Y' in queue table
#   end for row in results.fetchall():
#   END PROCESS COMPLETED JOBS
#
#   sleep(SLEEP_INTERVAL)
