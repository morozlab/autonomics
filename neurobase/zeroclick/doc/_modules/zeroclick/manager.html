[docs]def start_job(job, job_list, pipe_resources, mysql_session):
    try:
        #check if this job has unfinished dependencies in the queue
        depends = netutils.getTableObject("dependency", mysql_session)
        queue = netutils.getTableObject("queue", mysql_session)
        s = select([queue, depends], and_(queue.c.job_type==depends.c.depends_on, queue.c.finished=='N', depends.c.job_type==job.jobType, queue.c.project_id==job.pid))
        dependency = s.execute().fetchone()
        if(dependency is None):
            mysql_session.conn.execute("UPDATE queue SET started='N' WHERE project_id='" + str(job.pid) + "' AND job_id='" + str(job.jid) + "'")
            job.start()
            pipe_resources.giveTo(job)
            job_list.append(job)
            mysql_session.conn.execute("UPDATE queue SET started='Y', s_ts=CURRENT_TIMESTAMP() WHERE project_id='" + str(job.pid) + "' AND job_id='" + str(job.jid) + "'")
            #else:
            #print("Unsatisfied dependency for: " + job.jobName)

    except Exception as e:
        print(e)
        sys.stderr.write(e.message + "\n")
        sys.stderr.write("Error starting job: " + job.jobName + "\n")
        raise

    parser = argparse.ArgumentParser()
    parser.add_argument("--name", dest="myName", default=None)
    parser.add_argument("-u", "--user", dest="user", default=None)
    parser.add_argument("-mu", "--mail-user", dest="mailUser", default=None)
    parser.add_argument("-mh", "--mail-host", dest="mailHost", default="gmail.com")
    parser.add_argument("-du", "--database-user", dest="dbUser", default=None)
    parser.add_argument("-d", "--directory", dest="localDir", default="/srv/data2/pipeline/", help="Directory this manager should watch for new jobs.")
    parser.add_argument('--remote-dir', dest='remoteDir', default='/scratch/hpc/mcitar/')
    parser.add_argument("-q", "--queue", dest="queue", default="bio", help="Queue for submission of HPC jobs.")
    parser.add_argument("--sleep-interval", dest="mainLoopSleepInterval", default=settings.mainLoopSleepInterval, help="How often this manager should sleep between checking job statuses.")

    args = parser.parse_args()
    sys.stdout = Unbuffered(sys.stdout)
    #assign a name to this job manager
    if(args.myName is None):
        t = datetime.datetime.now()
        myName = "manager" + str(t.year) + str(t.month) + str(t.day) + str(t.hour) + str(t.minute) + str(t.second)
    else:
        myName = args.myName

    #initialize the daggum resources
    PIPE_RESOURCES = Resources()
    PIPE_RESOURCES.addResource("cpu", 61)
    PIPE_RESOURCES.addResource("a8r", 5)

    HPC_RESOURCES = Resources()
    HPC_RESOURCES.addResource("cpu", 1024)

    resources_at_location = {Locations.LOCAL: PIPE_RESOURCES, Locations.HPC: HPC_RESOURCES}

    SLEEP_INTERVAL = float(args.mainLoopSleepInterval)

    #get the pertinent credentials
    if(not args.user is None):
       passwd = raw_input("Enter HPC password:")
       settings.hpc_cred.update(args.user, passwd)

    if(not args.mailUser is None):
        mailPasswd = raw_input("Enter mail account password: ")
        settings.mail_cred.update(args.user, mailPasswd)
    if(not args.dbUser is None):
        dbPasswd = raw_input("Enter database passwod:")
        settings.db_cred.update(args.user, dbPasswd)

    #make one of them-there credentials objects
    hpcCred = settings.hpc_cred
    mailCred = settings.mail_cred

    #create a db session for doing cool stuffs for the good of all!
    session = netutils.DBSession("localhost", "zero_click", settings.db_cred.user, settings.db_cred.passwd)
    session.activateConnection()
    #list of jobs and such
    job_list = []
    finished = []
    #enter the loop in which you will stay for all enterity
    while(True):
        try:
            #get an object representing the queue table
            q = netutils.getTableObject("queue", session)
            #get the jobs that are not running
            s = q.select().where(q.c.started=='N')
            results = s.execute()
            for row in results.fetchall():
                #create a job for each job yet to be started
                job = jobConstructors[row.job_type](row)
                if(resources_at_location[job.location].hasEnoughFree(job.resources)):
                    start_job(job, job_list, resources_at_location[job.location], session)

            finished = []
            for j in job_list:
                stat = j.check()
                print(j.pn + " " + str(j.jid) + " " + j.jobType + ": " + str(stat))
                if(stat == JobState.FINISHED):
                    finished.append(j)
                    #complete the job, retreiving output if necessary and cleaning up temporary files
                    j.complete()
                    resources_at_location[j.location].takeFrom(j)

                    #update the 'finished' and f_ts fields of the queue
                    u = q.update().where(q.c.job_id==j.jid).values(f_ts=functions.current_timestamp(), finished='Y').execute()

            job_list = [j for j in job_list if not j in finished]


            for loc, resources in resources_at_location.items():
                print('--- Resources ---')
                print(resources.free)
                print(resources.totals)
                print('-----------------')

            gc.collect()
            print(gc.garbage)
            print("Sleeping for: " + str(SLEEP_INTERVAL))
            time.sleep(SLEEP_INTERVAL)
        except exc.OperationalError as e:
            if("MySQL server has gone away" in e.message):
                #re-initialize the session and re-enter the main loop
                print("MySQL server has gone away, restarting session.\n")
                session = netutils.DBSession("localhost", "zero_click", settings.db_cred.user, settings.db_cred.passwd)
                session.activateConnection()


if __name__ == '__main__':
    main()
